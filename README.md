# Perintah
    $ pkg update && upgrade
    $ pkg install git
    $ pkg install python
    $ pip install requests
    $ git clone https://github.com/RozhakXD/Vipig
    $ cd Vipig
    $ python Run.py

# Cara Update
    $ cd $HOME/Vipig
    $ git pull
    $ python Run.py
